632 struct aglUniformBuffer
633 {
634 aglUniformBuffer(aglShader* shader, aglUniformBufferSettings settings)
635 {
636 this->shader = shader;
637 this->settings = settings;
638
639 // UB creation
640
641
642#ifdef GRAPHICS_VULKAN
643 VkDeviceSize bufferSize = sizeof(T);
644
645 uniformBuffers.resize(MAX_FRAMES_IN_FLIGHT);
646 ubMemory.resize(MAX_FRAMES_IN_FLIGHT);
647 mappedUbs.resize(MAX_FRAMES_IN_FLIGHT);
648
649 for (int i = 0; i < MAX_FRAMES_IN_FLIGHT; ++i)
650 {
651 CreateBuffer(bufferSize, VK_BUFFER_USAGE_UNIFORM_BUFFER_BIT,
652 VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT | VK_MEMORY_PROPERTY_HOST_COHERENT_BIT,
653 uniformBuffers[i], ubMemory[i]);
654
655 vkMapMemory(GetDevice(), ubMemory[i], 0, bufferSize, 0, &mappedUbs[i]);
656 }
657#else
658
659 if (settings.binding >= 32)
660 {
661 throw new exception("Invalid binding.");
662 }
663
664 glGenBuffers(1, &ubo);
665 glBindBuffer(GL_UNIFORM_BUFFER, ubo);
666 glBufferData(GL_UNIFORM_BUFFER, sizeof(T), NULL, GL_STATIC_DRAW); // allocate 152 bytes of memory
667 glBindBuffer(GL_UNIFORM_BUFFER, 0);
668
669 glBindBufferRange(GL_UNIFORM_BUFFER, settings.binding, ubo, 0, sizeof(T));
670#endif
671 }
672
673 void Destroy()
674 {
675#ifdef GRAPHICS_VULKAN
676 for (size_t i = 0; i < MAX_FRAMES_IN_FLIGHT; i++)
677 {
678 vkDestroyBuffer(GetDevice(), uniformBuffers[i], nullptr);
679 vkFreeMemory(GetDevice(), ubMemory[i], nullptr);
680 }
681#else
682
683#endif
684 }

686 void Update(T data)
687 {
688#ifdef GRAPHICS_VULKAN

690#else
691 glBindBuffer(GL_UNIFORM_BUFFER, ubo);

699 void AttachToShader(aglShader* shader, u32 bindingIdx)
700 {
701#ifdef GRAPHICS_VULKAN
702 VkDescriptorSetLayoutBinding uboLayoutBinding{};
703 uboLayoutBinding.binding = bindingIdx;
704 uboLayoutBinding.descriptorType = VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER;
705 uboLayoutBinding.descriptorCount = 1;
706 uboLayoutBinding.stageFlags = settings.flags;
707 uboLayoutBinding.pImmutableSamplers = nullptr;
708
709 VkDescriptorPoolSize uboPoolSize{};
710
711 uboPoolSize.type = VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER;
712 uboPoolSize.descriptorCount = static_cast<u32>(MAX_FRAMES_IN_FLIGHT);
713
714 shader->AttachDescriptorPool(uboPoolSize);
715
716 CreateBinding(uboLayoutBinding);
717
718 for (int i = 0; i < MAX_FRAMES_IN_FLIGHT; ++i)
719 {
720 auto bufferInfo = new VkDescriptorBufferInfo;
721 bufferInfo->buffer = GetUniformBuffer(i);
722 bufferInfo->offset = 0;
723 bufferInfo->range = sizeof(T);
724
725 VkWriteDescriptorSet* descriptorWrite;
726
727
728 descriptorWrite = shader->CreateDescriptorSetWrite(i, bindingIdx);
729 descriptorWrite->descriptorType = VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER;
730 descriptorWrite->pBufferInfo = bufferInfo;
731
732 shader->AttachDescriptorWrite(descriptorWrite, i, bindingIdx);
733 }
734
735#else
736 unsigned index = glGetUniformBlockIndex(shader->id, settings.name.c_str());
737 glUniformBlockBinding(shader->id, index, settings.binding);

739
740#endif
